## (2) 排序國文成績, 由大到小

### index.js

```javascript
//----------------------------
// 引用db.js
//----------------------------
var myDB = require('./lib/db.js');


//----------------------------
// 篩選資料集內容的函式
//----------------------------
function readAllData(dataset){	
    return myDB.ref(dataset).orderByChild('國文').once('value')
        .then(snapshot => {
            if (!snapshot.exists()) {
                console.log("找不到資料");
            }else{
                //儲存結果
                results=[];				
			
                snapshot.forEach(childSnapshot => {
                    //某個學生的資料
                    var childKey = childSnapshot.key;
                    var childData = childSnapshot.val();

                    //儲存一位學生所有的成績資料					
                    items=[];
					
                    for(key in childData) {
                        var value = childData[key];
                        var d = {};
                        d[key] = value;
                        items.push(d);
                    }
					
                    //將學號及成績放入一個物件中
                    var t = {};
                    t[childKey] = items;
					
                    //加入結果
                    results.push(t);					
                })
				
                //以反向順序取出結果
                while(data = results.pop()){
                    console.log(data);				
                }
            }
        })
        .catch(error => {
            console.log('錯誤');
        });
}


//----------------------------
// 讀入資料
//----------------------------
readAllData('scores');
```


## (3) 由物件中的子物排序 (由學生所屬的學校, 再其內的縣市別排序)

// 引用db.js
//----------------------------
var myDB = require('./lib/db.js');

//----------------------------
// 讀入一筆資料的函式
//----------------------------
function readData(dataset){	
    return myDB.ref(dataset).orderByChild('school/city').once('value')
        .then(snapshot => {
            if (!snapshot.exists()) {
                console.log("找不到資料");
            }else{	
                snapshot.forEach(childSnapshot => {
                    var key = childSnapshot.key;
                    var value = childSnapshot.val();		
					
                    console.log(key);
                    console.log(value);
                });	
            }			
        })
        .catch(error => {
            console.log('錯誤');
        });
}

//----------------------------
// 讀入資料
//----------------------------
readData('students');
