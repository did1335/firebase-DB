### (1-2) index.js

```javascript
//----------------------------
// 引用db.js
//----------------------------
var myDB = require('./lib/db.js');

//----------------------------
// 讀入一個Collection的函式
//----------------------------
function readData(stuNo){    
    return new Promise(function(resolve, reject){
        myDB.ref('/students/' + stuNo).once('value').then(snapshot => {
            if (!snapshot.exists()) {               
                resolve({});  //回傳空資料
            }else{
                //---------------------------------
                // 取出快照內容
                //---------------------------------
                results={}; 
				
                snapshot.forEach(childSnapshot => {
                    key = childSnapshot.key;
                    var value = childSnapshot.val();
                    results[key] = value;
                });
                //---------------------------------
                	
                resolve(results);  //回傳資料
            }
        });
    });
}


//----------------------------
// 讀入資料
//----------------------------
readData('120001').then(results => {
    console.log(results.name);
    console.log(results.gender);
    console.log(results.type);
    console.log(results.school);
});
```



## (2) 讀一個Collection的資料   .ref('students')

### index.js

```javascript
//----------------------------
// 引用db.js
//----------------------------
var myDB = require('./lib/db.js');

//----------------------------
// 讀入一個Collection的函式
//----------------------------
function readData(collections){    
    return new Promise(function(resolve, reject){		
        myDB.ref(collections).once('value').then(snapshot => {				
            if (!snapshot.exists()) {
                //回傳空資料
                resolve([]);
            }else{					
                results=[]; 
				
                snapshot.forEach(childSnapshot => {
                    var obj={};
                    key = childSnapshot.key;
                    var value = childSnapshot.val();
                    obj[key] = value;
							
                    results.push(obj);
                });	
				
                //回傳讀出的資料		
                resolve(results);				
            }
        });	
    });
}


//----------------------------
// 讀入資料
//----------------------------
readData('students').then(results => {
    results.forEach(doc => {
        console.log(Object.keys(doc)[0]);
        console.log(Object.values(doc)[0]);
        console.log();
    });
});	
```
